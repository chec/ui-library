<Preview>

    {{
      components: {
        BaseNotification
      },
      data() {
        return {
          closed: false,
        };

      },
      template: `

          <BaseNotification
            v-if="!closed"
            :variant="variant"
            @closet="close()"
          >
            {{text}}
          </BaseNotification>
        </div>`
    }}

# Variants

<Preview>
  <Story name="Variants">
    {{
      components: {
        BaseNotification
      },
      data() {
        return {
          variants: ['success', 'error', 'warning', 'info'],
          closed: [],
        };
      },
      methods: {
        close(variant) {
          if (!this.isClosed(variant)) {
            this.closed.push(variant);
            setTimeout(() => {
              this.closed.splice(this.closed.indexOf(variant), 1);
            }, 3000)
          }
        },
        isClosed(variant) {
          return this.closed.includes(variant);
        },
        content(variant) {
          return `${variant.substring(0, 1).toUpperCase()}${variant.substring(1)} variant`;
        }
      },
      template: `
        <div class="mx-auto max-w-lg pt-6">
          <BaseNotification
            v-for="variant in variants"
            v-if="!isClosed(variant)"
            class="mb-2"
            :variant="variant"
            @close="close(variant)"
          >
            {{content(variant)}}
          </BaseNotification>
        </div>`
    }}
  </Story>
</Preview>
