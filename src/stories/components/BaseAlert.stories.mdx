          <BaseAlert

            :variant="variant"

            :show-icon="showIcon"

# Variants

<Preview>
  <Story name="Variants">
    {{
      components: {
        BaseAlert
      },
      data() {
        return {
          variants: ['success', 'error', 'warning', 'info'],
          closed: [],
        };
      },
      methods: {
        close(variant) {
          if (!this.isClosed(variant)) {
            this.closed.push(variant);
            setTimeout(() => {
              this.closed.splice(this.closed.indexOf(variant), 1);
            }, 3000)
          }
        },
        isClosed(variant) {
          return this.closed.includes(variant);
        },
        content(variant) {
          return `${variant.substring(0, 1).toUpperCase()}${variant.substring(1)} variant`;
        }
      },
      template: `
        <div class="mx-auto px-16 pt-6">
          <BaseAlert
            v-for="variant in variants"
            v-show="!isClosed(variant)"
            class="my-6"
            :variant="variant"
            @close="close(variant)"
          >
            {{content(variant)}}
          </BaseAlert>
        </div>`
    }}
  </Story>
</Preview>
