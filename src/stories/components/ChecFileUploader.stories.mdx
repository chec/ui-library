# Default:

<Preview>
  <Story name="Default">
    {{
      components: {
        ChecFileUploader,
      },
      data() {
        return {
          files: {},
        };
      },
      props: {
        endpoint: {
          default: text('endpoint', 'https://httpbin.org/post'),
        },
      },
      methods: {
        addedFile(file) {
          action('added-file');
          this.files[file.upload.uuid] = file;
        },
        removeFile(file) {
          action('remove-file');
          if (file.upload.progress < 100) {
            return;
          }
          const files = { ...this.files };
          delete files[file.upload.uuid];
          this.files = { ...files };
        },
        uploadingFile(file) {
          action('uploading-file');
          this.files[file.upload.uuid] = file;
        },
      },
      template: `
        <div class="p-16 flex justify-center w-full h-full bg-white py-20">
          <ChecFileUploader
            :files="files"
            @file-added="addedFile"
            @remove-file="removeFile"
            @file-uploading="uploadingFile"
            :endpoint="endpoint"
          />
        </div>`
    }}
  </Story>
</Preview>
