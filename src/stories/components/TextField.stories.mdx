# With action button

<Preview>
  <Story name="With action button">
    {{
      props: {
        inputValue: {
          default: text('Input value', 'Hello world')
        },
        label: {
          default: text('Label', 'Enter your greeting')
        },
        actionLabel: {
          default: text('Action label', 'Click me'),
        },
        tooltip: {
          default: text('Tooltip copy', 'Lorem Ipsum Dolor'),
        }
      },
      methods: {
        handleClick(e, value) {
          return action(`Text field action button was clicked with: ${value}`)();
        }
      },
      components: {
        TextField,
        ChecFormFieldActionPrepend
      },
      template: `
        <div class="py-16 flex justify-center">
          <div>
            <ChecFormFieldActionPrepend
              class="text-right"
              :label="actionLabel"
              :tooltip="tooltip"
            />
            <TextField
              :label="label"
              v-model="inputValue"
            />
            <ChecFormFieldActionPrepend
              class="text-right"
              :label="actionLabel"
              @action-click="handleClick"
            />
          </div>
        </div>
      `
    }}
  </Story>
</Preview>
